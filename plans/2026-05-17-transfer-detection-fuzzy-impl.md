```swift
// Domain/Models/DismissedTransferPair.swift

import Foundation

/// A user assertion that two specific transactions are NOT a transfer.
/// Permanent: detection skips any candidate pair covered by one of
/// these. Synced so the dismissal is respected on every device.
struct DismissedTransferPair: Codable, Sendable, Identifiable, Hashable {
  /// Deterministic, order-independent: the lower UUID string,
  /// `:` separator, then the higher — hashed into a stable UUID via
  /// `UUID.deterministic(from:)`. Same pair → same id on every device,
  /// so a re-dismissal upserts rather than duplicating.
  let id: UUID
  let transactionIds: Set<UUID>
  let dismissedAt: Date

  init(transactionIds: Set<UUID>, dismissedAt: Date) {
    self.transactionIds = transactionIds
    self.dismissedAt = dismissedAt
    self.id = Self.deterministicId(for: transactionIds)
  }

  func covers(_ first: UUID, _ second: UUID) -> Bool {
    transactionIds == [first, second]
  }

  static func deterministicId(for ids: Set<UUID>) -> UUID {
    let ordered = ids.map(\.uuidString).sorted().joined(separator: ":")
    return UUID.deterministic(from: "dismissed-transfer-pair:\(ordered)")
  }
}
```

```swift
// Domain/Models/UUID+Deterministic.swift

import CryptoKit
import Foundation

extension UUID {
  /// A stable UUID derived from `seed` (first 16 bytes of its SHA-256,
  /// RFC-4122 version/variant bits set). Same seed → same UUID on every
  /// device. Used for content-addressed records like
  /// `DismissedTransferPair`.
  static func deterministic(from seed: String) -> UUID {
    var digest = Array(SHA256.hash(data: Data(seed.utf8)).prefix(16))
    digest[6] = (digest[6] & 0x0F) | 0x40
    digest[8] = (digest[8] & 0x3F) | 0x80
    return UUID(uuid: (
      digest[0], digest[1], digest[2], digest[3],
      digest[4], digest[5], digest[6], digest[7],
      digest[8], digest[9], digest[10], digest[11],
      digest[12], digest[13], digest[14], digest[15]))
  }
}
```

```swift
// Backends/GRDB/ProfileSchema+TransferDetection.swift

import Foundation
import GRDB

extension ProfileSchema {
  /// v12 migration body. Adds the fuzzy-transfer-detection storage:
  ///
  /// - `transaction.transfer_suggestion_counterpart_id` (BLOB, nullable)
  ///   + `transaction.transfer_suggestion_suggested_at` (TEXT, nullable)
  ///   — the denormalised `TransferSuggestion`. NULL = no suggestion
  ///   (correct for every pre-v12 row).
  /// - `transaction.import_origin_kind` (TEXT, nullable, CHECK in
  ///   ('single','merged')). NULL = no import origin; 'single' = the
  ///   existing eight `import_origin_*` columns hold the lone origin
  ///   (so every legacy row reads back as `.single` with NULL kind
  ///   treated as single — see `TransactionRow+Mapping`); 'merged' =
  ///   existing eight columns hold the OUTGOING origin and the new
  ///   eight `import_origin_incoming_*` columns hold the INCOMING one.
  /// - Eight `import_origin_incoming_*` columns mirroring the existing
  ///   denormalised `ImportOrigin` shape for the merged incoming side.
  /// - `dismissed_transfer_pair` table — synced (CKSyncEngine) so a
  ///   "Not a transfer" decision is respected on every device.
  ///
  /// All additive; no table rebuild (no CHECK changes to existing
  /// columns). Legacy rows: every new column is NULL, which the mapping
  /// layer reads as "no suggestion / single import origin / not
  /// dismissed" — the correct semantics for everything pre-dating
  /// fuzzy detection.
  static func addTransferDetection(_ database: Database) throws {
    try database.execute(
      sql: """
        ALTER TABLE "transaction"
          ADD COLUMN transfer_suggestion_counterpart_id BLOB;
        ALTER TABLE "transaction"
          ADD COLUMN transfer_suggestion_suggested_at TEXT;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_kind TEXT
            CHECK (import_origin_kind IS NULL
                   OR import_origin_kind IN ('single', 'merged'));
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_raw_description TEXT;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_bank_reference TEXT;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_raw_amount TEXT;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_raw_balance TEXT;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_imported_at TEXT;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_import_session_id BLOB;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_source_filename TEXT;
        ALTER TABLE "transaction"
          ADD COLUMN import_origin_incoming_parser_identifier TEXT;

        CREATE TABLE dismissed_transfer_pair (
            id                     BLOB    NOT NULL PRIMARY KEY,
            record_name            TEXT    NOT NULL UNIQUE,
            transaction_id_a       BLOB    NOT NULL,
            transaction_id_b       BLOB    NOT NULL,
            dismissed_at           TEXT    NOT NULL,
            encoded_system_fields  BLOB
        ) STRICT;

        CREATE INDEX dismissed_pair_by_tx_a
            ON dismissed_transfer_pair(transaction_id_a);
        CREATE INDEX dismissed_pair_by_tx_b
            ON dismissed_transfer_pair(transaction_id_b);
        """)
  }
}
```

```swift
// Shared/TransferDetection/FuzzyTransferDetector.swift

import Foundation

/// Pure fuzzy candidate finder. No I/O. Reuses Extension A
/// (`transferDetectionValueLeg`) for the value-bearing leg. Callers
/// must run `CrossAccountTransferMerger` (Extension B) BEFORE this and
/// pass only the survivors, per the design's Acceptance section.
struct FuzzyTransferDetector: Sendable {
  /// ±3-day window, inclusive (design §Algorithm).
  static let windowSeconds: TimeInterval = 3 * 86_400

  /// Returns unordered candidate pairs (each tx appears at most once).
  func detect(
    newlyImported: [Transaction],
    existingNearby: [Transaction],
    isDismissed: (UUID, UUID) -> Bool
  ) -> [(Transaction, Transaction)] {
    let pool = newlyImported + existingNearby
    var consumed: Set<UUID> = []
    var result: [(Transaction, Transaction)] = []

    for tx in newlyImported {
      guard !consumed.contains(tx.id),
        let leg = tx.transferDetectionValueLeg,
        leg.type != .transfer,
        let accountId = leg.accountId
      else { continue }

      let matches = pool.compactMap { other -> (Transaction, TimeInterval)? in
        guard other.id != tx.id, !consumed.contains(other.id),
          let oleg = other.transferDetectionValueLeg,
          oleg.type != .transfer,
          let oAccount = oleg.accountId, oAccount != accountId,
          oleg.instrument == leg.instrument,
          oleg.quantity == -leg.quantity,
          !isDismissed(tx.id, other.id)
        else { return nil }
        let gap = abs(other.date.timeIntervalSince(tx.date))
        guard gap <= Self.windowSeconds else { return nil }
        return (other, gap)
      }

      // Closest-dated wins; ties broken by lower counterpart UUID
      // (deterministic — design §Algorithm).
      guard let best = matches.min(by: { lhs, rhs in
        lhs.1 != rhs.1 ? lhs.1 < rhs.1
          : lhs.0.id.uuidString < rhs.0.id.uuidString
      })?.0 else { continue }

      consumed.insert(tx.id)
      consumed.insert(best.id)
      result.append((tx, best))
    }
    return result
  }
}
```
